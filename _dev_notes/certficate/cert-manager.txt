Cluster uses cert-manager service
for PKI certification and rotation of internal certificate used by Linkerd
as well as ingress traffic to web clients
and outbound traffic to external databases

# CREATE TRUST ANCHOR CERTIFICATE:
kubectl create secret tls \
    linkerd-trust-anchor \
    --cert=ca.crt \
    --key=ca.key \
    --namespace=linkerd

from the monorepo root
run `step certificate create root.cluster.local ca.crt ca.key --profile root-ca --no-password --insecure`
